6) Write a C program using switch to display day of the week by entering the corresponding number. For example, if input is 1, it will display Sunday and so on.

#include<stdio.h>

  int day;

  //Input number
  printf("Enter a number (1-7): ");
  scanf("%d", &day);

  // Switch to display day
  switch(day) {
    case 1:
        printf("Sunday\n");
        break;

    case 2:
        printf("Monday\n");
        break;

    case 3:
        printf("Tuesday\n");
  }
